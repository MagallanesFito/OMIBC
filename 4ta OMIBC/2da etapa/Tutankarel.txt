void llenar()
{
    iterate(4)
    {
        if(frontIsClear)putbeeper();
        turnleft();
    }
    iterate(4)
    {
        if(frontIsClear)
        {
            move();
            if(notNextToABeeper)llenar();

            move();
            iterate(2)turnleft();
        }

    }
}
